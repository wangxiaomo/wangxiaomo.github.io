中文正则遇到了些问题,不过好在解决掉了. 之前想的投机的抓取方案被自己果断的给否掉了. 其实, 现在写的也是有点投机所在的…不过, 问题不大. 以后肯定会被自己大篇幅改掉的…

use utf8;
sub getSymbianList($){
    my $savefile = $data_dir . "symbian_list.txt";
    my ($content,) = @_;
    my $count = 0;
    
    my @lines = split("\n", $content);
    
    my $loopNum = scalar(@lines);
    my $i = 0;
    
    my $file_handler;
    open $file_handler, ">", $savefile;
    print $file_handler "[塞班系统智能手机关注排行]\n";
    
    while($i<$loopNum){
        unless($lines[$i++] =~ /塞班系统智能手机关注排行/){
            next;
        }
        print $i;
        
        $i += 3;
        while(1){
            last if $count == 8;
            $count++;
            
            $lines[$i] =~ /title =\"(.*?)\"/;
            my $name = $1;
            
            # 处理中文
            $name =~ s/([A-Za-z1-9])\s+([A-Za-z1-9])/$1\/$2/g;
            $name =~ s/\s+//g;
            $name =~ s/\// /g;
             
            print $file_handler "name:$name\n";
            $lines[$i] =~ /\(.*?)\<\/span\>/;
            print $file_handler "price:$1\n";
            $lines[$i] =~ /href=\"(.*?)\"/;
            my $url = $1;
            print $file_handler "url:$url\n";
            
            my $detail = get($url);
            my @detail_lines = split("\n", $detail);
            my $detailLoopNum = scalar(@detail_lines);
            my $j = 0;
            
            while($j<$detailLoopNum){
                unless($detail_lines[$j] =~ /alt=\"$name\"/){
                    $j++;
                    next;
                }
                $detail_lines[$j] =~ /src=\"(.*?)\"/;
                my $pic_src = $1;
                
                my $pic_file = $img_dir . $name . ".jpg";
                my $pic_data = get($pic_src);
                
                my $pic_handler;
                open $pic_handler, ">", $pic_file;
                binmode($pic_handler);
                print $pic_handler $pic_data;
                close $pic_handler;
    
                my $score_num = 0;
                while($j<$detailLoopNum){
                    last if $score_num == 2;
                    unless($detail_lines[$j] =~ /sinfo/){
                        $j++;
                        next;
                    }
                    $score_num++;
                    $detail_lines[$j] =~ /.*\>(.*?)\<\/b\>/;
                    print $file_handler "score:$1\n";
                    $j++;
                }
                last;
            }
            print $file_handler "\n";
            $i++;
        }
    }
    close($file_handler);
}
