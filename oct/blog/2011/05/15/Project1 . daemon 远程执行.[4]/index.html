Project1 . Daemon 远程执行.[4]

May 15th, 2011

写好webclient了以后.我们就要开始写socketserver了. python提供了两个socket模块.一个是通用的.另一个是方便用来server的.这里选取通用的socket模块.

#!/usr/bin/python3
# -*- coding: utf8 -*-

import config
import func
import sys,os,re
from socket import *

class SocketServer():
    def __init__(self):
        func.ignore()

    def run(self):
        self.sock = socket(AF_INET, SOCK_STREAM)
        addr = config.ADDR
        port = int(config.PORT)
        self.sock.bind((addr, port))
        self.sock.listen(5)

        while True:
            client,(c_addr,c_port) = self.sock.accept()
            try:           
                pid = os.fork()
                if pid > 0:
                    continue

                while True:
                    command = client.recv(256)
                    if command == 'quit' or command == 'exit':
                        client.send("Connection Closed\n".encode("utf8"))
                        client.close()
                        sys.exit(1)
                    command = command.decode("utf8").strip("\r\n")
                    if re.match(r'^cd', command):
                        wd = command.split(" ")[1]
                        os.chdir(wd)
                        wd += "\n"
                        ret = wd.encode("utf8")
                    else:
                        ret = os.popen(command).read().encode("utf8")
                    client.send(ret)
            except:
                pass

    def __del__(self):
        self.sock.close()

if __name__ == '__main__':
    '''
    obj = SocketServer()
    obj.run()
    '''
    sys.stderr.write("This Must Be Run as A Module\n")
    sys.exit(2)

这里需要注意的一个问题是.server是由core fork出来的.server在接收到连接时又fork.在ppid返回时会发生错误.所以我们把异常trap掉. 还有一个需要注意的是,socket的send和recv都是byte类型的.所以需要对string进行转换.

string -> byte          encode
byte->string            decode
