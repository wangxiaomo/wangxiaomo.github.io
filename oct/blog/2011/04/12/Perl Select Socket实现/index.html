Perl Select Socket实现

Apr 12th, 2011 

use strict;
use IO::Socket;
use IO::Select;

$|++;  #open autoflush
my $s = new IO::Socket::INET(
         LocalAddr   =>   'localhost',
         LocalPort    =>   1234,
         Listen         =>    10,
         Proto          =>    'tcp',
) or die $@;
# 创建一个select 目标
my $read_set = new IO::Select(); 
$read_set->add($s);

while(1){
    #IO::Select的select方法.第一个参数设置为select目标.
    #然后方法会一直检查可读目标.直到有可用句柄返回.
    #返回一个列表.第一个参数表示可读的句柄数组引用.
    my ($rh_set) = IO::Select->select($read_set, undef, undef);
    foreach my $rh (@$rh_set){
          # 如果当前可读的socket等于侦听socket.那么说明有请求进来.
          # 应该及时accept.把已经建立的socket句柄加入检查队列.
          if($rh == $s){
                my $ns = $rh->accept();
                $read_set->add($ns);
          }else{
          # 使用sysread读取数据.每次读取32字节.并且只处理这32字节.
          # 如果客户发送多于32字节.会分为几次进行处理.
          # 如果多个客户端同时发送.那么处理是无序的.
          # 避免使用<>方法来读取socket.
          # 因为<>是面向行的.perl IO对它进行了缓冲.IO::Select看不到这个缓冲.
                my $buf = undef;
                if(sysread($rh, $buf, 32)){
                       print $buf;
                }else{
                       #没有更多的数据.在队列中除去
                       $read_set->remove($rh);
                }
         }
 }

客户端:

use strict;
use IO::Socket;

my @childs;
# fork 3个进程同时发送数据
for( 1..3){
     my $child = fork;
     if($child){
           push(@childs, $child);
     }else{
           # 创建到server的连接
           my $sock = new IO::Socket::INET(
                    PeerAddr =>  'localhost',
                    PeerPort  =>  1234,
                    Proto       =>   'tcp',
           ) or die $@;
           for (1..10){
                    print $sock random() . "\n";
                    select(undef, undef, undef, 0.25);
                    # 每发送一次.就休眠0.25秒.
           }
           $sock->close or die $@;
           exit 0;
      }
for (@childs){
      my $tmp = waitpid($_, 0);
      print "Done with pid $tmp\n";
}
sub random{
      my @x = ( 0..9,'a'..'z','A'..'Z');
      join ' ',map { $x[int rand @x]} 1..49;
}
