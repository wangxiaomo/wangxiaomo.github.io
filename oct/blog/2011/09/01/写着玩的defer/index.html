写着玩的defer

Sep 1st, 2011 

#!/usr/bin/python
#-*- coding: utf-8 -*-

def passthru(args):
    """passthru function"""
    return args

class ErrBack(Exception):
    """ErrBack"""
    message = "TEST"

class Deferred:
    """My Deferred"""
    def __init__(self):
        """init defered"""
        self.callbacks = []
        self.result = None

    def addCallbacks(self, callback=None, errback=None):
        """addCallbacks"""
        assert callable(callback)
        assert errback == None or callable(errback)
        self.callbacks.append((callback, errback or passthru))

    def addCallback(self, callback):
        """addCallback"""
        self.addCallbacks(callback, None)

    def addErrback(self, errback):
        """addErrback"""
        self.addCallbacks(passthru, errback)   
        
    def addBoth(self, callback):
        """addBoth"""
        self.addCallbacks(callback, callback)

    def callback(self, result):
        """callback start into chain"""
        self._startRunCallbacks(result)

    def errback(self, fail=None):
        """errback start into chain"""
        if not isinstance(fail, Exception):
            fail = ErrBack()
        self._startRunCallbacks(fail)

    def _startRunCallbacks(self, result):
        """start to run callbacks"""
        self.result = result
        self._runCallbacks()

    def _runCallbacks(self):
        """run callbacks"""
        while self.callbacks:
            callback,errback = self.callbacks.pop(0)
            if not isinstance(self.result, Exception):
                func = callback
            else:
                func = errback
            try:
                self.result = func(self.result)
            except:
                self.result = ErrBack()

    def __str__(self):
        result = getattr(self, 'result', "NO CALLBACKS")
        if isinstance(result, Exception):
            result = result.message
        return result

    def showCallbacks(self, debug=False):
        if debug is True:
            print self.callbacks 

def main():
    def a(_):
        return "A"
    def b(_):
        raise Exception("Exception")

    d = Deferred()
    d.addCallbacks(a, b)
    d.showCallbacks(True)
    d.errback(True)
    print d
