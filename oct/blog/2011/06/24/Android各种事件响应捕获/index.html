Android各种事件响应捕获

Jun 24th, 2011 

//编辑文本框的按键事件.
myEdit.setOnKeyListener(new OnKeyListener(){
    public boolean onKey(View v, int keyCode, KeyEvent event){
         myEdit.setText("");
         return true;
    }
});
//编辑文本框的焦点事件.
myEdit.setOnFocusChangeListener(new OnFocusChangeListener(){
    public void onFocusChangeListener(View v, boolean hasFocus){
        Toast.makeTest(getApplicationContext(), myEdit.getText(), Toast.LENGTH_LONG);
    }
});
//多选框的选择事件.
cbl.setOnCheckedChangeListener(new OnCheckedChangeListener(){
    public void onCheckedChangeListener(CompoundButton buttonView, boolean isChecked){
         Toast.makeText(getApplicationContext(), cbl.isChecked()+"", Toast.LENGTH_LONG);
    }
});
//按钮的事件.
btn.setOnClickListener(new OnClickListener(){
    public void onClick(View v){
        Toast.makeText(getApplicationContext(), btn.getText(), Toast.LENGTH_LONG);
    }
});
