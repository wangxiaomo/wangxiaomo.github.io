#!/usr/bin/env python
# -*- coding: utf8 -*-

# Perl BUILDIN FUNCTION
def push(array, *value):
    '''
    push value to array
    '''
    if len(value) == 1:
        array.append(value)
    else:
        array.extend(value)

def pop(array):
    '''
    pop value from array
    '''
    return array.pop()

def unshift(array, *value):
    '''
    unshift value to array
    '''
    if len(value) == 1:
        array.insert(0, value)
    else:
        for i in value:
            array.insert(0, i)
        
def shift(array):
    '''
    shift value from array
    '''
    return array.pop(0)

上面的代码没什么复杂的. 唯一需要注意的就是参数位置的. Python中传递参数有4种形式. 1.普通传递. 2.默认参数传递. 3.传递.[未知数量参数] 4.**传递.[按dict形式传递.传参数时必须指明行参.] That’s all. 在oschina上看到别人频繁的使用str方法.自己试验了下.果然好用.

Python中关于拷贝和引用

May 9th, 2011

今天上毛邓的时候又看了看python.听课对我来说还是比较难得 .今天博客的重点就是关于python各种数据结构的拷贝复制.然后就是函数参数传递时引用与否吧. 普通变量是不存在引用问题的.这里普通变量指的就是字符串.数值等不可变的变量.他们是不存在引用问题的. 如果想要改变他们的值/ 最好以list的形式传入函数, 然后就是list.

names = ['apple', 'orange', 'banana']
bak1 = names
names[1] = 'xxxx'
print("names: " + str(names))
print("bak1: " + str(bak1))

bak2 = names[:]
names[1] = 'yyyy'
print("names: " + str(names))
print("bak2: " + str(bak2))

然后就是关于dict.

#深拷贝 浅拷贝.
#浅拷贝只拷贝一层.当第一层内有其他的引用时.不改变第二层引用..
info = {'name':'wxm', 'age':20}
bak1 = info
#这个只是单纯的引用.
bak2 = info.copy()
#这个是简单的浅拷贝.
from copy import deepcopy
bak3 = info.deepcopy()
#这个是深拷贝

这就是大概的内容了. 根据上面我们可以知道. 当list,dict作为函数参数传递时,传递的是引用,我们可以改变他们的值.当string,number作为参数传递时,我们是不能改变他们的值的.

#!/usr/bin/python
import pygame, sys
from pygame.locals import *

#初始化
pygame.init()
screen = pygame.display.set_mode((320,240), 0, 32)
# 第一个参数是大小.第三个参数是色彩

# 载入图片
bg_pic = 'bg.xcf'
mouse_pic = 'mouse.xcf'
background = pygame.image.load(bg_pic).convert()
mouseCur = pygame.image.load(mouse_pic).convert_alpha()
# convert转化为普通图片.convert_alpha转化为分层图片.即背景透明.

# 进入MainLoop显示界面.
while True:
   # 捕捉事件
   for event in pygame.event.get():
         if event.type == QUIT:
             pygame.quit()
             sys.exit()
   # 设置背景
   screen.blit(background, (0,0))
   
   # 设置鼠标所指为图片中心
   x,y = pygame.mouse.get_pos()
   x -= background.get_width()/2
   y -= mouseCur.get_height()/2
   
   screen.blit(mouseCur, (x, y))
   
   pygame.display.update()

Apr 28th, 2011

昨晚看小说看的太HIGH了. 早上起来就7:50了. 赶紧洗涮了就飞奔. 到了教室才发现忘了带眼镜了. 所以后半节课就果断翘课了. 昨晚清哥写代码写到很晚. 抽住了. 有女生求他写代码 . 他就写的更入迷了似的. 上学期课设的时候我做过这个. 不过代码丢了. 大概印象还是有的. 现在就把代码确认下. 大概思路先说下:

首先connect默认是阻塞连接.阻塞连接时会持续请求连接.
大概30秒或者更长的时间才会返回.
当我们做端口扫描的时候完全不可能等待那么长时间.
所以我们需要先把socket设置成非阻塞的.
那么connect就会瞬间返回.[我们可以设置connect时间.].
然后我们通过select来判断套接字是否可读写.
这样就可以了.

下面看代码.

#include "stdio.h"                                                  
#include "winsock2.h"                                                
#pragma comment(lib, "ws2_32.lib")             
                                                                                                                             
#define IP_ADDRESS "127.0.0.1"       
#define PORT 8080          
                                                                                                                             
int main(void){     
        WSADATA wsaData;  
        WSAStartup(MAKEWORD(2,0), &wsaData); 
        SOCKET client = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP); 
        //int timeOut = 2000; // 2 seconds    
        //setsockopt(client, SOL_SOCKET, SO_SNDTIMEO, (char *)&timeOut, sizeof(timeOut));    
        //setsockopt(client, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeOut, sizeof(timeOut));
        unsigned long ul = 1;   
        ioctlsocket(client, FIONBIO, (unsigned long *)&ul);   
        SOCKADDR_IN saddr;         
        saddr.sin_family = AF_INET; 
        saddr.sin_addr.s_addr = inet_addr(IP_ADDRESS);  
        saddr.sin_port = htons(PORT);    
        connect(client, (SOCKADDR *)&saddr, sizeof(SOCKADDR)); 
        struct timeval timeout;
        fd_set r;        
        FD_ZERO(&r);      
        FD_SET(client, &r);   
        timeout.tv_sec = 2;  
        timeout.tv_usec = 0;    
        int ret = select(0, 0, &r, 0, &timeout); 
        // TODO with ret  
        if(ret > 0){     
                // Socket Open    
        }else{                   
                // Socket Close
        }
        closesocket(client);
        WSACleanup();
        return 0;
}
